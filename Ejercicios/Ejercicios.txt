int ordenarVector(int vector[], int len)
{
   int i,j;
   int temp;

   for(i=1;i<len;i++)
   {
        temp = vector[i];
        j=i-1;

        while(j>=0 && temp<vector[j])
        {
            vector[j+1] = vector[j];
            j--;
        }

        vector[j+1]=temp;
    }

    return 0;
}
